const fs = require("fs/promises"); const path = require("path"); const slugify = require("slugify"); const crypto = require("crypto"); // ====================================== // CONFIG // ====================================== const CONFIG = { totalPages: 50000, outputDir: "output", domain: "https://example.com", brand: "Brightlane", ctaLink: "https://try.monday.com/m1dglo0ttcfr", affiliateDisclosure: "Affiliate Disclosure: We may earn commissions from qualifying purchases.", sitemapChunkSize: 5000, silos: [ "project management", "kanban software", "workflow automation", "AI productivity", "task management", "business OS", "team collaboration", "enterprise workflow", "startup tools", "agile planning" ], intents: [ "review", "pricing", "comparison", "alternatives", "best tools", "guide", "software", "platform" ], modifiers: [ "best", "top", "advanced", "AI powered", "enterprise", "free", "scalable", "modern", "next gen", "2026" ], audiences: [ "for startups", "for agencies", "for remote teams", "for enterprises", "for developers", "for marketing teams", "for operations", "for SMBs" ] }; // ====================================== // HELPERS // ====================================== function pick(arr) { return arr[Math.floor(Math.random() * arr.length)]; } function hash(input) { return crypto.createHash("md5").update(input).digest("hex").slice(0, 8); } function escapeHtml(str) { return str .replace(/&/g, "&") .replace(//g, ">") .replace(/"/g, """); } // ====================================== // CONTENT VARIATION ENGINE // ====================================== const intros = [ "Modern companies rely on {keyword} to streamline workflows.", "{keyword} has become essential for scaling operations.", "Teams use {keyword} to improve productivity and collaboration.", "{keyword} helps reduce operational bottlenecks." ]; const benefits = [ "Automates repetitive workflows", "Improves cross-team visibility", "Centralizes project execution", "Enhances operational efficiency", "Supports AI-assisted productivity" ]; const ctas = [ "Start Free Trial", "Try It Today", "Get Started", "Launch Your Workspace" ]; function spin(template, vars) { return template.replace(/\{(\w+)\}/g, (_, k) => vars[k] || ""); } function generateKeyword() { return [ pick(CONFIG.modifiers), pick(CONFIG.silos), pick(CONFIG.audiences) ].join(" "); } function buildContent(keyword, intent) { const intro = spin(pick(intros), { keyword }); const uniqueBenefits = [...benefits] .sort(() => Math.random() - 0.5) .slice(0, 4); return `

Why ${escapeHtml(keyword)} Matters

${escapeHtml(intro)}

Key Benefits

    ${uniqueBenefits.map(b => `
  • ${escapeHtml(b)}
    • `).join("")}

${escapeHtml(intent)} Insights

${escapeHtml(keyword)} enables organizations to improve execution speed, automate workflows, and reduce operational overhead.

${pick(ctas)}
`; } // ====================================== // INTERNAL LINKING // ====================================== function generateInternalLinks(currentSlug, pages) { const links = []; for (let i = 0; i < 5; i++) { const p = pick(pages); if (!p || p.slug === currentSlug) continue; links.push(`
  • ${escapeHtml(p.keyword)}
    • `); } return `

    Related Resources

      ${links.join("")}
    `; } // ====================================== // STRUCTURED DATA // ====================================== function schema(keyword) { return ` `; } // ====================================== // PAGE TEMPLATE // ====================================== function template({ title, description, slug, content, links }) { return ` ${escapeHtml(title)} ${schema(title)}

    ${escapeHtml(title)}

    Discover how ${escapeHtml(title)} improves modern workflows.

    ${pick(ctas)}

    ${CONFIG.affiliateDisclosure}

    ${content} ${links}
    © 2026 ${CONFIG.brand}
    `; } // ====================================== // SITEMAPS // ====================================== async function generateSitemaps(urls) { const chunks = []; for (let i = 0; i < urls.length; i += CONFIG.sitemapChunkSize) { chunks.push(urls.slice(i, i + CONFIG.sitemapChunkSize)); } const sitemapFiles = []; for (let i = 0; i < chunks.length; i++) { const xml = ` ${chunks[i] .map( url => ` ${url} ` ) .join("")} `; const file = `sitemap-${i}.xml`; await fs.writeFile(file, xml); sitemapFiles.push(file); } const index = ` ${sitemapFiles .map( f => ` ${CONFIG.domain}/${f} ` ) .join("")} `; await fs.writeFile("sitemap.xml", index); } // ====================================== // MAIN GENERATOR // ====================================== async function generate() { await fs.mkdir(CONFIG.outputDir, { recursive: true }); const urls = []; const pageRefs = []; for (let i = 0; i < CONFIG.totalPages; i++) { const keyword = generateKeyword(); const intent = pick(CONFIG.intents); const slug = slugify( `${keyword}-${intent}-${hash(keyword + i)}`, { lower: true, strict: true } ); pageRefs.push({ slug, keyword }); } for (let i = 0; i < pageRefs.length; i++) { const page = pageRefs[i]; const content = buildContent(page.keyword, pick(CONFIG.intents)); const links = generateInternalLinks(page.slug, pageRefs); const html = template({ title: page.keyword, description: `${page.keyword} review, pricing, features, and comparisons.`, slug: page.slug, content, links }); await fs.writeFile( path.join(CONFIG.outputDir, `${page.slug}.html`), html ); urls.push(`${CONFIG.domain}/${page.slug}.html`); if (i % 1000 === 0) { console.log(`Generated ${i}/${CONFIG.totalPages}`); } } await generateSitemaps(urls); console.log("DONE"); } generate().catch(console.error);
